Public addAttachment
addAttachment(attachment: QuoteAttachment, parentId: string)

Method returns an observable of array of string.

Example:

```typescript import { QuoteAttachment, AttachmentService } from '@apttus/ecommerce'; import { Observable } from 'rxjs/Observable';

  export class MyComponent implements OnInit{
       attachments: Array<string>;
       attachments$: Observable<Array<string>>;

       constructor(private attachmentService: AttachmentService){}

       addAttachment(attachmentObj: QuoteAttachment, parentId: string){
            this.attachmentService.addAttachment(attachmentObj, parentId).subscribe(res => attachments = res);
            // or
            attachments$ = this.attachmentService.addAttachment(attachmentObj, parentId);
       }
  }
 ```

Description

The price summary component displays the total price, promotion(s) applied, tax, sub total and action button(s) as per the need.

Usage

import { PriceSummaryModule } from '@apttus/elements';

@NgModule({
  imports: [PriceSummaryModule, ...]
})
export class AppModule {}

Implements

OnInit OnDestroy OnChanges

Example

<apt-price-summary
   [record]="cart"
   [form]="form"
   [spinner]="isLoading"
   [page]="'checkout'"
   (onSubmitOrder)="handleClick()"
   (onRequestQuote)="handleClick()"
></apt-price-summary>

        <div class="card-header">
            <h5 class="card-title mb-0">{{'DETAILS.PRICE_SUMMARY' | translate }}</h5>
        </div>
        <div class="card-body">
            <div class="d-flex justify-content-between">
                <span> {{'MANAGE_CART.CART_SUMMARY.TOTAL_PRICE' | translate }} </span>
                <span class="font-weight-bold">
                    <apt-price [record]="record" type="baseExtended"></apt-price>
                </span>
            </div>

            <!-- Promotion field for Manage Cart & Checkout pages which does require storefront.EnablePromotions check -->
            <div *ngIf="view?.storefront?.EnablePromotions" class="d-flex mt-2 justify-content-between">
                <span> {{'MANAGE_CART.CART_SUMMARY.PROMOTIONS_APPLIED' | translate }} </span>
                <span>{{view?.totalPromotions | localCurrency | async}}</span>
            </div>

            <!-- Tax field is required only on Quote Details, Create Proposal & Checkout pages -->
            <div class="d-flex mt-2 justify-content-between" *ngIf="view?.totalEstimatedTax != null && view?.showTaxLink">
                <span>
                    <button [popover]="taxPophover" [outsideClick]="true" class="btn btn-link text-capitalize p-0"
                        *ngIf="view?.totalEstimatedTax !== 0" data-toggle="popover" data-placement="auto bottom"
                        (click)="openEstimateTaxPopup()">
                        {{'MANAGE_CART.CART_SUMMARY.TAX' | translate }} </button>
                    <span *ngIf="view?.totalEstimatedTax === 0"> {{'MANAGE_CART.CART_SUMMARY.TAX' | translate }} </span>
                </span>
                <span>{{view?.totalEstimatedTax | localCurrency | async}}</span>
            </div>

            <hr class="mt-3 mb-2" />

            <div class="d-flex justify-content-between">
                <span> {{'MANAGE_CART.CART_SUMMARY.SUB_TOTAL' | translate }} </span>
                <span class="font-weight-bold">
                    <ng-container *ngIf="!view?.showTaxLink; else showNetPrice">
                        <apt-price [record]="record" type="listExtended"></apt-price>
                    </ng-container>
                    <ng-template #showNetPrice>
                        <apt-price [record]="record" type="net"></apt-price>
                    </ng-template>
                </span>
            </div>

            <!-- Few pages doesn't require any action button; in that case no need to show <hr> -->
            <hr class="mt-2 mb-3" *ngIf="page !== 'quote-details' && page !== 'order-details'" />

            <!-- Action buttons for Manage Cart page -->
            <div class="d-flex justify-content-end" *ngIf="page === 'manage-cart' && !view?.quote?.Id">
                <button *ngIf="view?.isLoggedIn && view?.storefront?.EnableRequestQuote" class="btn btn-link"
                    [routerLink]="['/Proposals/create']" [hidden]="view?.loading" data-style="zoom-in"
                    [disabled]="!(view?.lineItemCount > 0) || view?.hasErrors === true">
                    {{'COMMON.REQUEST_QUOTE' | translate }}
                </button>
                <button class="btn btn-primary" [routerLink]="['/cart']" [ladda]="view?.loading"
                    data-style="zoom-in" [disabled]="!(view?.lineItemCount > 0) || view?.hasErrors === true">
                    {{'MANAGE_CART.CART_SUMMARY.CHECKOUT' | translate }}
                </button>
            </div>

            <div class="d-flex justify-content-end" *ngIf="page === 'manage-cart' && view?.quote?.Id">
                <button class="btn btn-link mr-2" (click)="openDiscardChageModals()" [hidden]="view?.loading"
                    data-style="zoom-in" [disabled]="!(view?.lineItemCount > 0) || view?.hasErrors === true">
                    {{'MANAGE_CART.CART_SUMMARY.DISCARD_CHANGES' | translate }}
                </button>
                <button class="btn btn-primary px-2" [routerLink]="['/Proposals/create']" [ladda]="view?.loading"
                    data-style="zoom-in" [disabled]="!(view?.lineItemCount > 0) || view?.hasErrors === true">
                    {{'MANAGE_CART.CART_SUMMARY.FINALIZE_AND_SUBMIT' | translate }}
                </button>
            </div>

            <!-- Action buttons for Checkout page -->
            <div *ngIf="page === 'checkout'">
                <div *ngIf="!view?.accountId && !view?.isLoggedIn; else showButton"
                    tooltip="{{'CART.DEFAULT_ACC_NOT_AVAILABLE' | translate}}">
                    <button class="btn btn-primary btn-block" [ladda]="view?.loading" data-style="zoom-in"
                        (click)="submitOrder()" [disabled]="true">
                        {{ 'COMMON.PLACE_ORDER' | translate }}
                    </button>
                </div>
                <ng-template #showButton>
                    <button class="btn btn-primary btn-block" [ladda]="view?.loading" data-style="zoom-in"
                        (click)="submitOrder()" [disabled]="!form.valid || !(view?.lineItemCount > 0)">
                        <span *ngIf="paymentState != 'PAYNOW'">{{ 'COMMON.PLACE_ORDER' | translate }}</span>
                        <span *ngIf="paymentState === 'PAYNOW'">{{ 'COMMON.PLACE_ORDER_PAY' | translate }}</span>
                    </button>
                </ng-template>
            </div>
            <div *ngIf="page === 'paymentForOrder'">
                <button class="btn btn-primary btn-block" [ladda]="view?.loading" data-style="zoom-in"

                    MAKE PAYMENT
                </button>
            </div>

            <!-- Action buttons for Create Proposal page -->
            <div class="d-flex justify-content-end" *ngIf="page === 'create-proposal'">

                    {{'COMMON.CANCEL' | translate }}
                </button>
                <button class="btn btn-primary" (click)="requestQuote()" *ngIf="!view?.quote?.Id"

                <button class="btn btn-link" (click)="discardChangesModal.hide()"> {{ 'COMMON.CANCEL' | translate }}
                </button>

                    {{ 'MANAGE_CART.DISCARD_CHANGES_MODAL.DISCARD_CHANGES' | translate }} </button>

Description

The Address component allows users to enter address details. The details include street name, city name, country code, state code and zip code. When a user tries to add a new address or update an existing address associated with his account, he can fill in details like street, city name and zip in the respective text boxes provided.

Usage

```typescript import { AddressModule } from '@apttus/elements';

@NgModule({ imports: [AddressModule, ...] }) export class AppModule {} ```

<div *ngIf="record">
  <ng-container *ngIf="!readonly; else read">
    <div class="mb-3">
      <label [for]="getId('street')">Address</label>
      <input type="text" class="form-control" [id]="getId('street')" placeholder="1234 Main St" required [(ngModel)]="record[street]"
        name="street" #address="ngModel" [ngClass]="{ 'is-invalid': address.invalid && (address.touched || address.dirty || form.submitted) }">
      <div class="invalid-feedback">
        Please enter your shipping address.
      </div>
    </div>
    
    <div class="mb-3">
      <label [for]="getId('city')">City</label>
      <input type="text" class="form-control" [id]="getId('city')" placeholder="City" required [(ngModel)]="record[city]"
        name="city" #cityTemplate="ngModel" [ngClass]="{ 'is-invalid': cityTemplate.invalid && (cityTemplate.touched || cityTemplate.dirty || form.submitted) }">
      <div class="invalid-feedback">
        Please enter your city.
      </div>
    </div>
    
    <div class="row">
      <div class="col-md-9 mb-3">
        <apt-dependent-picklist name="dependentPicklist" [fieldMethod]="fieldService" [field]="stateCode" [parentField]="countryCode" [(ngModel)]="countryState" (ngModelChange)="countryStateChange($event)"></apt-dependent-picklist>
      </div>
      <div class="col-md-3 mb-3">
        <label [for]="getId('zip')">Zip</label>
        <input type="text" class="form-control" [id]="getId('zip')" required [(ngModel)]="record[postalCode]"
          name="postal-code" #zipcode="ngModel" [ngClass]="{ 'is-invalid': zipcode.invalid && (zipcode.touched || zipcode.dirty || form.submitted) }">
        <div class="invalid-feedback">
          Zip code required.
        </div>
      </div>
    </div>
  </ng-container>
  <ng-template #read>
    <address>
      {{record[street]}}
      <br> {{record[city]}}, {{record[state]}} {{record[postalCode]}}
      <br> {{record[country]}}
    </address>
  </ng-template>
</div>

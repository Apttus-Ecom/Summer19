Description

Output field component is used to create a label value paired output that can be displayed as an editable field with an optional popover that shows related information about the field.

Usage

import { OutputFieldModule } from '@apttus/elements';

@ngModule({
  imports: [OutputFieldModule, ...]
})

Implements

OnChanges OnInit

Example

<apt-output-field
[record]="quote"
[field]="CreatedDate"
layout="inline"
[editable]="false"
></apt-output-field>

<span class="d-flex" [class.mb-2]="!labelOnly" [class.align-items-baseline]="layout === 'inline'" [class.flex-column]="layout === 'stacked'" [class.inline]="layout === 'inline'" *ngIf="metadata$ | async as metadata">
  <ng-container *ngTemplateOutlet="template; context: { ref: getLookupFieldsObjectRef(metadata) }"></ng-container>
  <ng-template let-lookupFieldsObjectRef="ref" #template>
    <dt [ngClass]="labelClass">
      {{label$ | async}}
    </dt>
    <dd *ngIf="!labelOnly" class="m-0 position-relative"  [class.px-1]="layout === 'inline'" [class.py-1]="layout === 'stacked'" [class.empty]="!getValue(metadata)" [ngClass]="valueClass" [ngSwitch]="true">
      <ng-container *ngSwitchCase="metadata.type === 'date' || metadata.type === 'datetime'">{{getValue(metadata) | date:getDateFormat(metadata)}}</ng-container>
  
      <ng-container *ngSwitchCase="metadata.type === 'currency'">{{getValue(metadata) | localCurrency | async}}</ng-container>
  
      <ng-container *ngSwitchCase="metadata.type === 'address'">
        <apt-address [value]="getValue(metadata)" [readonly]="true"></apt-address>
      </ng-container>
  
      <ng-container *ngSwitchCase="metadata.type === 'double'">
        {{getValue(metadata) | number:'1.0-' + metadata.scale}}
      </ng-container>

      <ng-container *ngSwitchDefault>
        <ng-container *ngIf="getDetailPagePath(lookupFieldsObjectRef) && !showQuickView; else nonLink">
          <a href="javascript:void(0)" [routerLink]="getDetailPagePath(lookupFieldsObjectRef)" [innerHTML]="getValue(metadata)" target="_blank"></a>
        </ng-container>

          <span
            [ngClass]="showQuickView ? 'btn-link popoverLink' : ''"

            containerClass="lookupPopover"
            [outsideClick]="true"
            [placement]="onShown()"
            [innerHTML]="getValue(metadata)"></span>
        </ng-template>
      </ng-container>
      <i

        (onShown)="hidePopover()"
      ></i>
    </dd>
    <ng-template #lookupPopTemplate>
      <div class="popHeader p-3 mb-1" [ngClass]="expanded ? 'bg-dark text-white' : 'bg-white text-dark'">

          <ng-container *ngIf="getDetailPagePath(lookupFieldsObjectRef); else noLinkPopoverTitle">

        </div>
        <div class="fieldSection container-fluid py-2">

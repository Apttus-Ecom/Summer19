Description

The change configuration modal is used to create a modal for terminating one or many assets. To open a terminate modal the AssetModalService must be used.

Usage

import { AssetModalService } from '@apttus/elements';

@Component({
  ...
})
export class AppComponent {

  private lineItems: Array<AssetLineItem>;

  constructor(private assetModalService: AssetModalService) {}

  openChangeConfigModal() {
    this.assetModalService.openChangeConfigurationModal(lineItems[0], lineItems);
  }
}

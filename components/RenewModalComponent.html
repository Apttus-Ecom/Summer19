Description

The renew modal is used to create a modal for renewing one or many assets. To open a renew modal the AssetModalService must be used.

Preview

Usage

import { AssetModalService } from '@apttus/elements';

@Component({
  ...
})
export class AppComponent {

  private lineItems: Array<AssetLineItem>;

  constructor(private assetModalService: AssetModalService) {}

  openRenewModal() {
    this.assetModalService.openRenewModal(lineItems[0], lineItems);
  }
}

Description

The compare table is a table of product features with a product compare card as each column header.

Implements

OnInit

Example

<apt-compare-table [products$]="productsObservable$"></apt-compare-table>

<table class="table" [style.min-width]="((products$ | async)?.length > 5) ? '1200px' : widths[(products$ | async)?.length]">
  <thead>
    <tr>
      <th></th>
      <th scope="col" *ngFor="let product of products$ | async">
        <apt-compare-product-card
          [product]="product"
        ></apt-compare-product-card>
      </th>
    </tr>
  </thead>
  <tbody>
    <ng-container *ngFor="let featureSet of featureSets">
      <tr>
        <th class="featureSetCell py-2" [attr.colspan]="(products$ | async)?.length + 1">
          {{featureSet.Name}}
        </th>
      </tr>
      <tr *ngFor="let feature of featuresInSet(featureSet)">
        <th class="bg-light">{{feature.Name}}</th>
        <td class="bg-white" *ngFor="let product of products$ | async">{{featureValue(product, feature)}}</td>
      </tr>
    </ng-container>
  </tbody>
</table>

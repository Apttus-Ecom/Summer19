projects/apttus/elements/src/lib/input-field/input-field.component.ts

Description

The Input field component displays the field values as a radio button, checkbox, dropdown, a text area or a label value, depending upon the field type.

Usage

import { InputFieldModule } from '@apttus/elements';

@NgModule({
  imports: [InputFieldModule, ...]
})
export class AppModule {}

Implements

ControlValueAccessor OnChanges OnDestroy

Example

<apt-input-field
  [(ngModel)]="myModel"
  (ngModelChange)="handleChange($event)"
  [field]="fieldApiName"
  [entity]="entityApiName"
  [label]="label"
  [readonly]="isReadOnly"
  [allowEmpty]="allowEmpty"
  [disabledValues]="disabledValues"
  [hiddenValues]="hiddenValues"
  [inline]="isInline"
  [small]="isSmall"
></apt-input-field>

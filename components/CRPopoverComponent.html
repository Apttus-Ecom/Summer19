projects/apttus/elements/src/lib/constraint-popover/constraint-popover.component.ts

Description

The constraint rule popover component creates a popover on an element based on the constraint rule list and product passed to it.

Preview

Usage

import { ConstraintPopoverModule } from '@apttus/elements';

@NgModule({
  imports: [ConstraintPopoverModule, ...]
})
export class AppModule {}

<button
   [popover]="popTemplate"
   container="body"
   [outsideClick]="true"
   containerClass="CRPopover"
   triggers="mouseenter:mouseleave focus:blur"

   Add to Cart
</button>

Description

Product drawer component is used to create a toggleable drawer on the bottom of the screen that displays currently selected products and their associated batch actions.

Preview

Usage

import { ProductDrawerModule } from '@apttus/elements';

@NgModule({
  imports: [ProductDrawerModule, ...]
})
export class AppModule {}

Implements

OnInit OnChanges OnDestroy

<apt-product-drawer [products]="products"></apt-product-drawer>

<div class="bannerWrapper " [class.hiddenBelow]='!(drawerIsOpen$ | async)'>
  <div class="body container-fluid" *ngIf="products$ | async as products">
    <div class="tab px-3" (click)="toggleDrawer()">
      <span><span class="d-none d-md-inline"> {{'PRODUCT_CARD.SELECTED_PRODUCTS' | translate}} </span> ({{products?.length}})</span>
      <apt-super-chevron direction="{{!(drawerIsOpen$ | async) ? 'up' : 'down'}}"></apt-super-chevron>
    </div>
    <div class="row h-100">
      <div class="col-lg-10 col-md-9 col-12">
        <div class="d-flex justify-content-center flex-column h-100" *ngIf="products && products?.length > 0">
          <ngx-slick class="carousel card-deck mx-md-4" [config]="slideConfig">
            <div ngxSlickItem *ngFor="let product of products">
              <apt-selected-product-card [product]="product"></apt-selected-product-card>
            </div>
          </ngx-slick>
        </div>
      </div>
      <div class="col-lg-2 col-md-3 col-12 my-md-2 pb-md-3 mh-100 batchActionWrapper">
        <apt-batch-action (actionFired)="onAction($event)"></apt-batch-action>
      </div>
    </div>
  </div>
</div>
